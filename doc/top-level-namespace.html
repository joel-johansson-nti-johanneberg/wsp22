post('/login') do
  if session[:timeLogged] == nil
    session[:timeLogged] = 0
  end
  logTime =  timeChecker(session[:timeLogged])
  session[:timeLogged] = Time.now.to_i
  if logTime
    username = params[:username]
    password = params[:password]
    login_post(username,password)
  else
    redirect('/showlogin')
  end
end
